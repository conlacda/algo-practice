Implement in python
```python
N,M = [int(i) for i in input().split()]

maxtrix = []
visited = []
for i in range(N):
    inp = input()
    maxtrix.append(inp)
    visited.append([])
    for j in range(len(inp)):
        if inp[j] == '#':
            visited[-1].append(True)
        else:
            visited[-1].append(False)
def explore(i,j):
    if not visited[i][j]:
        visited[i][j] = True
        if 0<=i-1<N and 0<=j<M and not visited[i-1][j]:
            # print(i,j,'->',i-1,j)
            explore(i-1, j)
        if 0<=i+1<N and 0<=j<M and not visited[i+1][j]:
            # print(i,j,'->',i+1,j)
            explore(i+1, j)
        if 0<=i<N and 0<=j-1<M and not visited[i][j-1]:
            # print(i,j,'->',i,j-1)
            explore(i, j-1)
        if 0<=i<N and 0<=j+1<M and not visited[i][j+1]:
            # print(i,j,'->',i,j+1)
            explore(i, j+1)
scc = 0
for i in range(N):
    for j in range(M):
        if not visited[i][j]:
            explore(i,j)
            scc+=1
print(scc)
```

#include<bits/stdc++.h>
 
using namespace std;
typedef long long ll;
const ll MOD = 1e9+7;
 
int N,M;
string s;
vector<vector<bool>> visited;
void explore(int i, int j){
    if (!visited[i][j]){
        visited[i][j] = true;
        if (0<=i-1 && i-1 < N && 0<=j && j<M && !visited[i-1][j]){
            explore(i-1,j);
        }
        if (0<=i+1 && i+1 < N && 0<=j && j<M && !visited[i+1][j]){
            explore(i+1,j);
        }
        if (0<=j-1 && j-1 < M && 0<=i && i<N && !visited[i][j-1]){
            explore(i,j-1);
        }
        if (0<=j+1 && j+1 < M && 0<=i && i<N && !visited[i][j+1]){
            explore(i,j+1);
        }
    }
}
int main(){
    cin >> N;
    cin >> M;
    for (int i=0;i<N;i++){
        cin >> s;
        vector<bool> x;
        for (int j=0;j<M;j++){
            if (s[j] == '#'){
                x.push_back(true);
            } else {
                x.push_back(false);
            }
        }
        visited.push_back(x);
    }
    int room=0;
    for (int i=0;i<N; i++){
        for (int j=0;j<M;j++){
            if (!visited[i][j]){
                explore(i,j);
                room ++;
            }
        }
    }
    cout << room;
    return 0;
}
